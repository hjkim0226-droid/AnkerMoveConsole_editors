66/*****************************************************************************
67 * Initialize
68 *****************************************************************************/
69void Initialize() {
70 if (g_hwnd) return;
71
72 // Initialize GDI+
73 GdiplusStartupInput gdiplusStartupInput;
74 Status gdipStatus = GdiplusStartup(&g_gdiplusToken, &gdiplusStartupInput, NULL);
75 if (gdipStatus != Ok) {
76 return;
77 }
78
79 // Register window class
80 WNDCLASSEXW wc = {0};
81 wc.cbSize = sizeof(WNDCLASSEXW);
82 wc.style = CS_HREDRAW | CS_VREDRAW;
83 wc.lpfnWndProc = WndProc;
84 wc.hInstance = GetModuleHandle(NULL);
85 wc.hCursor = LoadCursor(NULL, IDC_ARROW);
86 wc.hbrBackground = NULL;
87 wc.lpszClassName = L"DMenuUIWindow";
88 ATOM classAtom = RegisterClassExW(&wc);
89 if (classAtom == 0 && GetLastError() != ERROR_CLASS_ALREADY_EXISTS) {
90 GdiplusShutdown(g_gdiplusToken);
91 g_gdiplusToken = 0;
92 return;
93 }
94
95 // Create window (initially hidden) - NOTE: NO WS_EX_NOACTIVATE, we WANT focus
96 g_hwnd = CreateWindowExW(
97 WS_EX_TOPMOST | WS_EX_TOOLWINDOW,
98 L"DMenuUIWindow",
99 L"",
100 WS_POPUP,
101 0, 0, WINDOW_WIDTH, WINDOW_HEIGHT,
102 NULL, NULL, GetModuleHandle(NULL), NULL
103 );
104
105 if (!g_hwnd) {
106 GdiplusShutdown(g_gdiplusToken);
107 g_gdiplusToken = 0;
108 return;
109 }
110}

126/*****************************************************************************
127 * ShowMenu
128 *****************************************************************************/
129void ShowMenu(int x, int y) {
130 if (!g_hwnd) Initialize();
131 if (!g_hwnd) return;
132
133 g_action = ACTION_NONE;
134 g_hoverIndex = -1;
135
136 // Position near mouse
137 RECT workArea;
138 SystemParametersInfo(SPI_GETWORKAREA, 0, &workArea, 0);
139
140 int posX = x - WINDOW_WIDTH / 2;
141 int posY = y - 10; // Slightly above cursor
142
143 if (posX < workArea.left) posX = workArea.left;
144 if (posY < workArea.top) posY = workArea.top;
145 if (posX + WINDOW_WIDTH > workArea.right) posX = workArea.right - WINDOW_WIDTH;
146 if (posY + WINDOW_HEIGHT > workArea.bottom) posY = workArea.bottom - WINDOW_HEIGHT;
147
148 SetWindowPos(g_hwnd, HWND_TOPMOST, posX, posY, WINDOW_WIDTH, WINDOW_HEIGHT, SWP_SHOWWINDOW);
149 ShowWindow(g_hwnd, SW_SHOW);
150
151 // Take focus - this is key to intercepting keyboard input
152 SetForegroundWindow(g_hwnd);
153 SetFocus(g_hwnd);
154
155 g_visible = true;
156 g_showTime = std::chrono::steady_clock::now(); // Record show time for grace period
157}

190/*****************************************************************************
191 * HitTest
192 *****************************************************************************/
193static int HitTest(int x, int y) {
194 if (x < PADDING || x > WINDOW_WIDTH - PADDING) return -1;
195
196 int itemY = PADDING;
197 for (int i = 0; i < MENU_ITEM_COUNT; i++) {
198 if (y >= itemY && y < itemY + ITEM_HEIGHT) {
199 return i;
200 }
201 itemY += ITEM_HEIGHT;
202 }
203 return -1;
204}

206/*****************************************************************************
207 * Draw
208 *****************************************************************************/
209static void Draw(HDC hdc) {
210 Graphics g(hdc);
211 g.SetSmoothingMode(SmoothingModeAntiAlias);
212 g.SetTextRenderingHint(TextRenderingHintClearTypeGridFit);
213
214 // Background with rounded corners
215 SolidBrush bgBrush(COLOR_BG);
216 Pen borderPen(COLOR_BORDER, 1);
217
218 // Simple rectangle for now
219 g.FillRectangle(&bgBrush, 0, 0, WINDOW_WIDTH, WINDOW_HEIGHT);
220 g.DrawRectangle(&borderPen, 0, 0, WINDOW_WIDTH - 1, WINDOW_HEIGHT - 1);
221
222 // Menu items
223 FontFamily fontFamily(L"Segoe UI");
224 Font keyFont(&fontFamily, 11, FontStyleBold, UnitPixel);
225 Font labelFont(&fontFamily, 11, FontStyleRegular, UnitPixel);
226 SolidBrush textBrush(COLOR_TEXT);
227 SolidBrush keyBrush(COLOR_KEY);
228 SolidBrush hoverBrush(COLOR_ITEM_HOVER);
229
230 int itemY = PADDING;
231 for (int i = 0; i < MENU_ITEM_COUNT; i++) {
232 RECT itemRect = { PADDING, itemY, WINDOW_WIDTH - PADDING, itemY + ITEM_HEIGHT };
233
234 // Hover background
235 if (i == g_hoverIndex) {
236 g.FillRectangle(&hoverBrush, itemRect.left, itemRect.top,
237 itemRect.right - itemRect.left, itemRect.bottom - itemRect.top);
238 }
239
240 // Key indicator [A]
241 wchar_t keyText[4];
242 swprintf(keyText, 4, L"[%c]", MENU_ITEMS[i].key);
243 g.DrawString(keyText, -1, &keyFont, PointF((REAL)(PADDING + 4), (REAL)(itemY + 5)), &keyBrush);
244
245 // Label
246 g.DrawString(MENU_ITEMS[i].label, -1, &labelFont,
247 PointF((REAL)(PADDING + 40), (REAL)(itemY + 6)), &textBrush);
248
249 itemY += ITEM_HEIGHT;
250 }
251}

253/*****************************************************************************
254 * WndProc
255 *****************************************************************************/
256static LRESULT CALLBACK WndProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam) {
257 switch (msg) {
258 case WM_PAINT: {
259 PAINTSTRUCT ps;
260 HDC hdc = BeginPaint(hwnd, &ps);
261 Draw(hdc);
262 EndPaint(hwnd, &ps);
263 return 0;
264 }
265
266 case WM_MOUSEMOVE: {
267 int x = GET_X_LPARAM(lParam);
268 int y = GET_Y_LPARAM(lParam);
269 int newHover = HitTest(x, y);
270 if (newHover != g_hoverIndex) {
271 g_hoverIndex = newHover;
272 InvalidateRect(hwnd, NULL, FALSE);
273 }
274 return 0;
275 }
276
277 case WM_LBUTTONDOWN: {
278 int x = GET_X_LPARAM(lParam);
279 int y = GET_Y_LPARAM(lParam);
280 int index = HitTest(x, y);
281 if (index >= 0 && index < MENU_ITEM_COUNT) {
282 g_action = MENU_ITEMS[index].action;
283 HideMenu();
284 }
285 return 0;
286 }
287
288 case WM_KEYDOWN: {
289 // Handle key shortcuts
290 switch (wParam) {
291 case 'A':
292 case 'a':
293 g_action = ACTION_ALIGN;
294 HideMenu();
295 break;
296 case 'T':
297 case 't':
298 g_action = ACTION_TEXT;
299 HideMenu();
300 break;
301 case 'K':
302 case 'k':
303 g_action = ACTION_KEYFRAME;
304 HideMenu();
305 break;
306 case 'C':
307 case 'c':
308 g_action = ACTION_COMP;
309 HideMenu();
310 break;
311 case VK_ESCAPE:
312 g_action = ACTION_CANCELLED;
313 HideMenu();
314 break;
315 }
316 return 0;
317 }
318
319 case WM_CHAR:
320 // Consume char events to prevent beeps
321 return 0;
322
323 case WM_ACTIVATE:
324 // WM_ACTIVATE is more reliable than WM_KILLFOCUS for detecting outside clicks
325 if (LOWORD(wParam) == WA_INACTIVE) {
326 // Check if within grace period (ignore focus loss right after showing)
327 // Only cancel if no action was already set (prevents overwriting click selection)
328 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
329 std::chrono::steady_clock::now() - g_showTime).count();
330 if (elapsed > FOCUS_GRACE_PERIOD_MS && g_action == ACTION_NONE) {
331 // Window is being deactivated (clicked outside)
332 g_action = ACTION_CANCELLED;
333 HideMenu();
334 }
335 }
336 return 0;
337
338 case WM_KILLFOCUS:
339 // Backup: also handle kill focus (with grace period check)
340 // Only cancel if no action was already set (prevents overwriting click selection)
341 {
342 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>(
343 std::chrono::steady_clock::now() - g_showTime).count();
344 if (elapsed > FOCUS_GRACE_PERIOD_MS && g_action == ACTION_NONE) {
345 g_action = ACTION_CANCELLED;
346 HideMenu();
347 }
348 }
349 return 0;
350 }
351
352 return DefWindowProc(hwnd, msg, wParam, lParam);
353}
