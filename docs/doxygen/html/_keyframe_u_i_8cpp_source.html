26// Extract a float value from JSON string like "key": 123.45
27static float JsonGetFloat(const wchar_t* json, const wchar_t* key, float defaultVal = 0.0f) {
28 wchar_t searchKey[64];
29 swprintf_s(searchKey, L"\"%s\"", key);
30
31 const wchar_t* pos = wcsstr(json, searchKey);
32 if (!pos) return defaultVal;
33
34 // Find the colon after the key
35 pos = wcschr(pos, L':');
36 if (!pos) return defaultVal;
37 pos++;
38
39 // Skip whitespace
40 while (*pos == L' ' || *pos == L'\t' || *pos == L'\n' || *pos == L'\r') pos++;
41
42 // Parse the number
43 return (float)_wtof(pos);
44}

46// Extract a string value from JSON like "key": "value"
47static void JsonGetString(const wchar_t* json, const wchar_t* key, wchar_t* out, size_t outSize) {
48 out[0] = L'\0';
49 wchar_t searchKey[64];
50 swprintf_s(searchKey, L"\"%s\"", key);
51
52 const wchar_t* pos = wcsstr(json, searchKey);
53 if (!pos) return;
54
55 // Find the colon
56 pos = wcschr(pos, L':');
57 if (!pos) return;
58 pos++;
59
60 // Skip whitespace
61 while (*pos == L' ' || *pos == L'\t' || *pos == L'\n' || *pos == L'\r') pos++;
62
63 // Expect opening quote
64 if (*pos != L'"') return;
65 pos++;
66
67 // Copy until closing quote
68 size_t i = 0;
69 while (*pos && *pos != L'"' && i < outSize - 1) {
70 out[i++] = *pos++;
71 }
72 out[i] = L'\0';
73}

79// Convert After Effects speed/influence to bezier control points
80// Formula based on AE's KeyframeEase system:
81// avgSpeed = valueChange / duration
82// normalizedSpeed = speed / avgSpeed
83// P1.x = outInfluence / 100
84// P1.y = P1.x * normalizedOutSpeed
85// P2.x = 1 - (inInfluence / 100)
86// P2.y = 1 - (1 - P2.x) * normalizedInSpeed
87static void ConvertAEToBezier(
88 float outSpeed, float outInfluence,
89 float inSpeed, float inInfluence,
90 float avgSpeed,
91 KeyframeUI::KeyframeType outType,
92 KeyframeUI::KeyframeType inType,
93 KeyframeUI::VelocityCurve& curve)
94{
95 // Handle edge case: no value change (avgSpeed = 0)
96 // In this case, show a flat line
97 if (fabs(avgSpeed) < 0.0001f) {
98 curve.p0_x = 0.25f;
99 curve.p0_y = 0.25f;
100 curve.p1_x = 0.75f;
101 curve.p1_y = 0.75f;
102 return;
103 }
104
105 // Handle special keyframe types
106 // Hold keyframe: horizontal line (instant value jump at start/end)
107 if (outType == KeyframeUI::KEYFRAME_HOLD) {
108 // Out from hold: horizontal line at start
109 curve.p0_x = 0.33f;
110 curve.p0_y = 0.0f; // Flat at bottom
111 curve.p1_x = 0.67f;
112 curve.p1_y = 1.0f; // Jump to top at end
113 return;
114 }
115 if (inType == KeyframeUI::KEYFRAME_HOLD) {
116 // In to hold: horizontal line at end
117 curve.p0_x = 0.33f;
118 curve.p0_y = 0.0f; // Start at bottom
119 curve.p1_x = 0.67f;
120 curve.p1_y = 1.0f; // Jump at end
121 return;
122 }
123
124 // Linear keyframe: diagonal line (constant velocity)
125 if (outType == KeyframeUI::KEYFRAME_LINEAR && inType == KeyframeUI::KEYFRAME_LINEAR) {
126 curve.p0_x = 0.33f;
127 curve.p0_y = 0.33f;
128 curve.p1_x = 0.67f;
129 curve.p1_y = 0.67f;
130 return;
131 }
132
133 // Bezier keyframe: use speed/influence to calculate control points
134 // Clamp influence to valid range
135 outInfluence = max(0.1f, min(100.0f, outInfluence));
136 inInfluence = max(0.1f, min(100.0f, inInfluence));
137
138 // Normalized speeds (relative to average)
139 float normalizedOutSpeed = outSpeed / avgSpeed;
140 float normalizedInSpeed = inSpeed / avgSpeed;
141
142 // Handle linear out with bezier in (or vice versa)
143 if (outType == KeyframeUI::KEYFRAME_LINEAR) {
144 normalizedOutSpeed = 1.0f; // Linear = same as average speed
145 outInfluence = 33.33f;
146 }
147 if (inType == KeyframeUI::KEYFRAME_LINEAR) {
148 normalizedInSpeed = 1.0f;
149 inInfluence = 33.33f;
150 }
151
152 // First control point (affects "out" from first keyframe)
153 curve.p0_x = outInfluence / 100.0f;
154 curve.p0_y = curve.p0_x * normalizedOutSpeed;
155
156 // Second control point (affects "in" to second keyframe)
157 curve.p1_x = 1.0f - (inInfluence / 100.0f);
158 curve.p1_y = 1.0f - ((1.0f - curve.p1_x) * normalizedInSpeed);
159
160 // Clamp Y values to reasonable range (-0.5 to 1.5 for overshoot effects)
161 curve.p0_y = max(-0.5f, min(1.5f, curve.p0_y));
162 curve.p1_y = max(-0.5f, min(1.5f, curve.p1_y));
163}

165// Convert bezier control points back to AE speed/influence
166// Reverse of ConvertAEToBezier
167static void ConvertBezierToAE(
168 const KeyframeUI::VelocityCurve& curve,
169 float avgSpeed,
170 float& outSpeed, float& outInfluence,
171 float& inSpeed, float& inInfluence)
172{
173 // Influence from X coordinates
174 outInfluence = curve.p0_x * 100.0f;
175 inInfluence = (1.0f - curve.p1_x) * 100.0f;
176
177 // Clamp influence to valid range (AE requires 0.1-100)
178 outInfluence = max(0.1f, min(100.0f, outInfluence));
179 inInfluence = max(0.1f, min(100.0f, inInfluence));
180
181 // Calculate normalized speeds from control points
182 // normalizedOutSpeed = P1.y / P1.x (when P1.x != 0)
183 float normalizedOutSpeed = 1.0f; // Default: linear
184 if (fabs(curve.p0_x) > 0.001f) {
185 normalizedOutSpeed = curve.p0_y / curve.p0_x;
186 }
187
188 // normalizedInSpeed = (1 - P2.y) / (1 - P2.x) (when P2.x != 1)
189 float normalizedInSpeed = 1.0f; // Default: linear
190 if (fabs(1.0f - curve.p1_x) > 0.001f) {
191 normalizedInSpeed = (1.0f - curve.p1_y) / (1.0f - curve.p1_x);
192 }
193
194 // If avgSpeed is 0 or very small, use default speed of 1.0
195 // This happens when value change is 0 (no animation)
196 float safeAvgSpeed = (fabs(avgSpeed) < 0.0001f) ? 1.0f : avgSpeed;
197
198 // Clamp normalized speeds first (prevent huge values)
199 normalizedOutSpeed = max(-100.0f, min(100.0f, normalizedOutSpeed));
200 normalizedInSpeed = max(-100.0f, min(100.0f, normalizedInSpeed));
201
202 // Convert back to actual speeds
203 outSpeed = normalizedOutSpeed * safeAvgSpeed;
204 inSpeed = normalizedInSpeed * safeAvgSpeed;
205
206 // Clamp speeds to prevent NaN/inf in script (reasonable range: 0-10000)
207 outSpeed = max(0.0f, min(10000.0f, outSpeed));
208 inSpeed = max(0.0f, min(10000.0f, inSpeed));
209
210 // Final NaN and infinity check
211 // Finite numbers satisfy: (x - x == 0) and (x == x)
212 if (outSpeed != outSpeed || (outSpeed - outSpeed) != 0.0f) outSpeed = safeAvgSpeed;
213 if (inSpeed != inSpeed || (inSpeed - inSpeed) != 0.0f) inSpeed = safeAvgSpeed;
214 if (outInfluence != outInfluence || (outInfluence - outInfluence) != 0.0f) outInfluence = 33.33f;
215 if (inInfluence != inInfluence || (inInfluence - inInfluence) != 0.0f) inInfluence = 33.33f;
216}

365// Draw slot icon
366// iconType: 0=Empty/Number, 1=Star, 2=Circle, 3=Wave, 4=Diamond
367void DrawSlotIcon(Graphics& graphics, int iconType, float cx, float cy, float size, const Color& color) {
368 SolidBrush brush(color);
369 Pen pen(color, 1.5f);
370 float r = size / 2.0f;
371
372 switch (iconType) {
373 case 1: { // Star (5-pointed)
374 PointF starPoints[10];
375 for (int i = 0; i < 10; i++) {
376 float angle = -90.0f + i * 36.0f;
377 float rad = angle * 3.14159f / 180.0f;
378 float dist = (i % 2 == 0) ? r : r * 0.4f;
379 starPoints[i].X = cx + cosf(rad) * dist;
380 starPoints[i].Y = cy + sinf(rad) * dist;
381 }
382 graphics.FillPolygon(&brush, starPoints, 10);
383 break;
384 }
385 case 2: { // Circle
386 graphics.FillEllipse(&brush, cx - r * 0.7f, cy - r * 0.7f, r * 1.4f, r * 1.4f);
387 break;
388 }
389 case 3: { // Wave (sine wave symbol)
390 GraphicsPath path;
391 float waveR = r * 0.8f;
392 for (int i = 0; i <= 20; i++) {
393 float t = (float)i / 20.0f;
394 float x = cx - waveR + t * waveR * 2.0f;
395 float y = cy + sinf(t * 3.14159f * 2.0f) * (r * 0.5f);
396 if (i == 0) path.StartFigure();
397 else {
398 float prevT = (float)(i - 1) / 20.0f;
399 float prevX = cx - waveR + prevT * waveR * 2.0f;
400 float prevY = cy + sinf(prevT * 3.14159f * 2.0f) * (r * 0.5f);
401 path.AddLine(prevX, prevY, x, y);
402 }
403 }
404 Pen wavePen(color, 2.0f);
405 graphics.DrawPath(&wavePen, &path);
406 break;
407 }
408 case 4: { // Diamond
409 PointF diamondPoints[4] = {
410 PointF(cx, cy - r),
411 PointF(cx + r * 0.7f, cy),
412 PointF(cx, cy + r),
413 PointF(cx - r * 0.7f, cy)
414 };
415 graphics.FillPolygon(&brush, diamondPoints, 4);
416 break;
417 }
418 default:
419 break; // No icon for 0
420 }
421}

423// Draw mini bezier curve in a slot button
424void DrawMiniBezier(Graphics& graphics, const KeyframeUI::VelocityCurve& curve, int x, int y, int size, bool active) {
425 int padding = 4;
426 int drawX = x + padding;
427 int drawY = y + padding;
428 int drawSize = size - padding * 2;
429
430 // Background
431 SolidBrush bgBrush(active ? Color(255, 40, 60, 40) : Color(255, 30, 30, 35));
432 graphics.FillRectangle(&bgBrush, x, y, size, size);
433
434 // Draw curve
435 Pen curvePen(active ? COLOR_CURVE : Color(180, 74, 207, 255), 1.5f);
436
437 // Bezier curve points (normalized 0-1 to screen space)
438 std::vector<PointF> points;
439 for (int i = 0; i <= 20; i++) {
440 float t = (float)i / 20.0f;
441
442 // Cubic bezier: P = (1-t)^3*P0 + 3*(1-t)^2*t*P1 + 3*(1-t)*t^2*P2 + t^3*P3
443 float px = (1-t)*(1-t)*(1-t)*0.0f +
444 3*(1-t)*(1-t)*t*curve.p0_x +
445 3*(1-t)*t*t*curve.p1_x +
446 t*t*t*1.0f;
447 float py = (1-t)*(1-t)*(1-t)*0.0f +
448 3*(1-t)*(1-t)*t*curve.p0_y +
449 3*(1-t)*t*t*curve.p1_y +
450 t*t*t*1.0f;
451
452 // Map to screen (Y inverted)
453 float screenX = drawX + px * drawSize;
454 float screenY = drawY + (1.0f - py) * drawSize;
455 points.push_back(PointF(screenX, screenY));
456 }
457
458 // Draw curve segments
459 for (size_t i = 1; i < points.size(); i++) {
460 graphics.DrawLine(&curvePen, points[i-1], points[i]);
461 }
462
463 // Border
464 Pen borderPen(active ? COLOR_PRESET_ACTIVE : COLOR_BORDER, 1);
465 graphics.DrawRectangle(&borderPen, x, y, size, size);
466}

622// Helper: Parse a single keyframe pair object from JSON
623static void ParseSingleKeyframePair(const wchar_t* json, KeyframePairInfo& pair) {
624 // Extract property name
625 JsonGetString(json, L"propName", pair.info.propName, 128);
626 JsonGetString(json, L"propMatchName", pair.info.propMatchName, 128);
627
628 // Extract keyframe indices
629 pair.info.keyIndex1 = (int)JsonGetFloat(json, L"keyIndex1", 1);
630 pair.info.keyIndex2 = (int)JsonGetFloat(json, L"keyIndex2", 2);
631
632 // Extract times and values
633 pair.info.time1 = JsonGetFloat(json, L"time1", 0.0f);
634 pair.info.time2 = JsonGetFloat(json, L"time2", 1.0f);
635 pair.info.value1 = JsonGetFloat(json, L"value1", 0.0f);
636 pair.info.value2 = JsonGetFloat(json, L"value2", 0.0f);
637
638 // Extract easing values
639 pair.info.outSpeed = JsonGetFloat(json, L"outSpeed", 0.0f);
640 pair.info.outInfluence = JsonGetFloat(json, L"outInfluence", 33.33f);
641 pair.info.inSpeed = JsonGetFloat(json, L"inSpeed", 0.0f);
642 pair.info.inInfluence = JsonGetFloat(json, L"inInfluence", 33.33f);
643
644 // Sanitize parsed values (prevent NaN/inf)
645 auto sanitizeFloat = [](float& val, float defaultVal, float minVal, float maxVal) {
646 if (val != val || (val - val) != 0.0f) val = defaultVal; // NaN/inf check
647 val = max(minVal, min(maxVal, val));
648 };
649 sanitizeFloat(pair.info.outSpeed, 0.0f, 0.0f, 10000.0f);
650 sanitizeFloat(pair.info.outInfluence, 33.33f, 0.1f, 100.0f);
651 sanitizeFloat(pair.info.inSpeed, 0.0f, 0.0f, 10000.0f);
652 sanitizeFloat(pair.info.inInfluence, 33.33f, 0.1f, 100.0f);
653
654 // Extract keyframe types (1=linear, 2=bezier, 3=hold)
655 int outTypeInt = (int)JsonGetFloat(json, L"outType", 2);
656 int inTypeInt = (int)JsonGetFloat(json, L"inType", 2);
657 pair.info.outType = (KeyframeUI::KeyframeType)outTypeInt;
658 pair.info.inType = (KeyframeUI::KeyframeType)inTypeInt;
659
660 // Get average speed
661 pair.avgSpeed = JsonGetFloat(json, L"avgSpeed", 0.0f);
662
663 // If avgSpeed not provided, calculate it
664 if (pair.avgSpeed == 0.0f) {
665 float duration = pair.info.time2 - pair.info.time1;
666 float valueChange = pair.info.value2 - pair.info.value1;
667 if (fabs(duration) > 0.0001f) {
668 pair.avgSpeed = fabs(valueChange / duration);
669 }
670 }
671
672 // Sanitize avgSpeed (prevent NaN/inf from breaking script)
673 if (pair.avgSpeed != pair.avgSpeed || (pair.avgSpeed - pair.avgSpeed) != 0.0f) {
674 pair.avgSpeed = 1.0f; // Default to 1.0 for invalid values
675 }
676 pair.avgSpeed = max(0.0f, min(10000.0f, pair.avgSpeed));
677
678 // Convert AE easing to bezier control points
679 ConvertAEToBezier(
680 pair.info.outSpeed,
681 pair.info.outInfluence,
682 pair.info.inSpeed,
683 pair.info.inInfluence,
684 pair.avgSpeed,
685 pair.info.outType,
686 pair.info.inType,
687 pair.curve
688 );
689
690 pair.isMiddleKeyframe = false; // Will be set later based on context
691}

693void SetKeyframeInfo(const wchar_t* infoJson) {
694 if (!infoJson || wcslen(infoJson) == 0) {
695 g_hasKeyframeInfo = false;
696 g_numKeyframePairs = 0;
697 g_multiViewMode = false;
698 return;
699 }
700
701 // Reset state
702 g_numKeyframePairs = 0;
703 g_currentPairIndex = 0;
704 g_multiViewMode = false;
705
706 // Check if this is a JSON array (starts with '[')
707 const wchar_t* ptr = infoJson;
708 while (*ptr == L' ' || *ptr == L'\t' || *ptr == L'\n' || *ptr == L'\r') ptr++;
709
710 if (*ptr == L'[') {
711 // Parse JSON array - multiple keyframe pairs
712 ptr++; // Skip '['
713
714 while (*ptr && g_numKeyframePairs < MAX_KEYFRAME_PAIRS) {
715 // Skip whitespace
716 while (*ptr == L' ' || *ptr == L'\t' || *ptr == L'\n' || *ptr == L'\r' || *ptr == L',') ptr++;
717
718 if (*ptr == L']') break; // End of array
719
720 if (*ptr == L'{') {
721 // Find the matching closing brace
722 const wchar_t* objStart = ptr;
723 int braceCount = 1;
724 ptr++;
725
726 while (*ptr && braceCount > 0) {
727 if (*ptr == L'{') braceCount++;
728 else if (*ptr == L'}') braceCount--;
729 ptr++;
730 }
731
732 if (braceCount == 0) {
733 // Extract this object as a substring
734 size_t objLen = ptr - objStart;
735 wchar_t* objStr = new wchar_t[objLen + 1];
736 wcsncpy_s(objStr, objLen + 1, objStart, objLen);
737 objStr[objLen] = L'\0';
738
739 // Parse this object
740 ParseSingleKeyframePair(objStr, g_keyframePairs[g_numKeyframePairs]);
741
742 // Mark middle keyframes (not first, not last will be determined later)
743 g_numKeyframePairs++;
744
745 delete[] objStr;
746 }
747 } else {
748 ptr++; // Skip unknown character
749 }
750 }
751
752 // Mark middle keyframes (keyframes that appear in both pairs)
753 // For pairs: [K1-K2, K2-K3, K3-K4], K2 and K3 are middle keyframes
754 for (int i = 0; i < g_numKeyframePairs; i++) {
755 if (i > 0) {
756 // Check if this pair's first keyframe matches previous pair's second
757 if (g_keyframePairs[i].info.keyIndex1 == g_keyframePairs[i-1].info.keyIndex2) {
758 g_keyframePairs[i].isMiddleKeyframe = true;
759 }
760 }
761 }
762
763 g_multiViewMode = (g_numKeyframePairs > 1);
764
765 } else if (*ptr == L'{') {
766 // Single object - parse directly
767 ParseSingleKeyframePair(infoJson, g_keyframePairs[0]);
768 g_numKeyframePairs = 1;
769 g_multiViewMode = false;
770 } else {
771 // Invalid JSON
772 g_hasKeyframeInfo = false;
773 return;
774 }
775
776 // Initialize with first pair
777 if (g_numKeyframePairs > 0) {
778 g_keyframeInfo = g_keyframePairs[0].info;
779 g_currentCurve = g_keyframePairs[0].curve;
780 g_avgSpeed = g_keyframePairs[0].avgSpeed;
781 g_hasKeyframeInfo = true;
782 g_currentPreset = PRESET_CUSTOM;
783 } else {
784 g_hasKeyframeInfo = false;
785 }
786
787 // Reset loadRequested flag (load operation complete)
788 g_result.loadRequested = false;
789
790 // Redraw if visible
791 if (g_hwnd && g_isVisible) {
792 InvalidateRect(g_hwnd, NULL, TRUE);
793 }

846// Evaluate cubic bezier at parameter t
847PointF EvalCubicBezier(float t, float p0x, float p0y, float p1x, float p1y,
848 float p2x, float p2y, float p3x, float p3y) {
849 float u = 1.0f - t;
850 float tt = t * t;
851 float uu = u * u;
852 float uuu = uu * u;
853 float ttt = tt * t;
854
855 PointF result;
856 result.X = uuu * p0x + 3 * uu * t * p1x + 3 * u * tt * p2x + ttt * p3x;
857 result.Y = uuu * p0y + 3 * uu * t * p1y + 3 * u * tt * p2y + ttt * p3y;
858 return result;
859}

861// Integrate velocity curve using Simpson's rule
862float IntegrateVelocityCurve(const KeyframeUI::VelocityCurve& curve, float t) {
863 const int N = 100;
864 float h = t / N;
865 float sum = 0;
866
867 for (int i = 0; i <= N; i++) {
868 float ti = (float)i * h;
869 // Evaluate bezier Y at this t (velocity)
870 PointF pt = EvalCubicBezier(ti, 0, 0, curve.p0_x, curve.p0_y,
871 curve.p1_x, curve.p1_y, 1, 1);
872 float vi = pt.Y;
873
874 if (i == 0 || i == N) sum += vi;
875 else if (i % 2 == 0) sum += 2 * vi;
876 else sum += 4 * vi;
877 }
878
879 return (h / 3.0f) * sum;
880}

883// Graph padding constant for overshoot visualization
884static const float GRAPH_Y_PADDING = 0.2f; // 20% padding above and below

886// Helper: Map curve Y value (with overshoot) to screen Y
887inline float CurveYToScreen(float curveY, int graphY, int graphH) {
888 // curveY range: -0.2 to 1.2 (with padding)
889 // Screen: top = 1.2, bottom = -0.2
890 float normalizedY = (1.0f + GRAPH_Y_PADDING - curveY) / (1.0f + GRAPH_Y_PADDING * 2);
891 return graphY + normalizedY * graphH;
892}

894void DrawBezierCurve(Graphics& graphics, const KeyframeUI::VelocityCurve& curve,
895 int x, int y, int w, int h) {
896 // Draw the velocity curve as a bezier with overshoot support
897 std::vector<PointF> points;
898 const int segments = 50;
899
900 for (int i = 0; i <= segments; i++) {
901 float t = (float)i / segments;
902 PointF pt = EvalCubicBezier(t, 0, 1, curve.p0_x, 1 - curve.p0_y,
903 curve.p1_x, 1 - curve.p1_y, 1, 0);
904 // Map to graph coordinates with padding for overshoot
905 float px = x + pt.X * w;
906 // pt.Y is inverted (0=bottom, 1=top in curve space)
907 float curveY = 1.0f - pt.Y; // Convert back to curve Y
908 float py = CurveYToScreen(curveY, y, h);
909 points.push_back(PointF(px, py));
910 }
911
912 // Draw curve
913 Pen curvePen(COLOR_CURVE, 2.5f);
914 if (points.size() > 1) {
915 graphics.DrawLines(&curvePen, points.data(), (INT)points.size());
916 }
917
918 // Draw control point handles (with overshoot support)
919 float p0_screenX = x + curve.p0_x * w;
920 float p0_screenY = CurveYToScreen(curve.p0_y, y, h);
921 float p1_screenX = x + curve.p1_x * w;
922 float p1_screenY = CurveYToScreen(curve.p1_y, y, h);
923
924 // Endpoint screen positions (0 and 1 in curve space)
925 float startY = CurveYToScreen(0.0f, y, h); // Bottom endpoint
926 float endY = CurveYToScreen(1.0f, y, h); // Top endpoint
927
928 // Lines from endpoints to handles
929 Pen handleLinePen(COLOR_CURVE_DIM, 1);
930 graphics.DrawLine(&handleLinePen, (REAL)x, startY, p0_screenX, p0_screenY);
931 graphics.DrawLine(&handleLinePen, (REAL)(x + w), endY, p1_screenX, p1_screenY);
932
933 // Handle circles
934 const float handleRadius = 6.0f;
935 Color h0Color = g_handleHover[0] ? COLOR_HANDLE_HOVER : COLOR_HANDLE;
936 Color h1Color = g_handleHover[1] ? COLOR_HANDLE_HOVER : COLOR_HANDLE;
937
938 SolidBrush h0Brush(h0Color);
939 SolidBrush h1Brush(h1Color);
940
941 graphics.FillEllipse(&h0Brush,
942 p0_screenX - handleRadius, p0_screenY - handleRadius,
943 handleRadius * 2, handleRadius * 2);
944 graphics.FillEllipse(&h1Brush,
945 p1_screenX - handleRadius, p1_screenY - handleRadius,
946 handleRadius * 2, handleRadius * 2);
947
948 // Start/end points
949 SolidBrush endpointBrush(COLOR_TEXT);
950 const float endpointRadius = 4.0f;
951 graphics.FillEllipse(&endpointBrush,
952 (REAL)x - endpointRadius, startY - endpointRadius,
953 endpointRadius * 2, endpointRadius * 2);
954 graphics.FillEllipse(&endpointBrush,
955 (REAL)(x + w) - endpointRadius, endY - endpointRadius,
956 endpointRadius * 2, endpointRadius * 2);
957}
