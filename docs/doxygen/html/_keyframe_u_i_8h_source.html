14// Velocity curve presets
15enum VelocityPreset {
16 PRESET_LINEAR = 0, // Constant velocity
17 PRESET_EASE_IN, // Accelerating (slow to fast)
18 PRESET_EASE_OUT, // Decelerating (fast to slow)
19 PRESET_EASE_IN_OUT, // Slow-fast-slow
20 PRESET_EASE_OUT_IN, // Fast-slow-fast (bounce-like)
21 PRESET_CUSTOM // User-defined curve
22};
23
24// Bezier control points for velocity curve
25// Normalized to [0,1] range
26struct VelocityCurve {
27 float p0_x, p0_y; // First control point (after start)
28 float p1_x, p1_y; // Second control point (before end)
29};
30
31// Keyframe interpolation types
32enum KeyframeType {
33 KEYFRAME_LINEAR = 1, // Linear interpolation
34 KEYFRAME_BEZIER = 2, // Bezier (eased)
35 KEYFRAME_HOLD = 3 // Hold (step/instant)
36};
37
38// Keyframe info from After Effects
39struct KeyframeInfo {
40 wchar_t propName[128]; // Property display name
41 wchar_t propMatchName[128]; // Property match name
42 int keyIndex1; // First selected keyframe index
43 int keyIndex2; // Second selected keyframe index
44 float time1, time2; // Keyframe times
45 float value1, value2; // Keyframe values
46 // Current easing
47 float outSpeed, outInfluence; // Out ease at key1
48 float inSpeed, inInfluence; // In ease at key2
49 // Keyframe types
50 KeyframeType outType; // Interpolation type at key1 (out)
51 KeyframeType inType; // Interpolation type at key2 (in)
52};
53
54// Result after panel closes
55struct KeyframeResult {
56 bool cancelled = false; // True if ESC or clicked outside
57 bool applied = false; // True if easing should be applied
58 bool loadRequested = false; // True if Load button pressed (reload keyframe info)
59 VelocityPreset preset = PRESET_LINEAR;
60 VelocityCurve customCurve;
61
62 // Calculated AE KeyframeEase values
63 float outSpeed = 0.0f; // Speed at first keyframe
64 float outInfluence = 33.33f; // Influence at first keyframe (%)
65 float inSpeed = 0.0f; // Speed at second keyframe
66 float inInfluence = 33.33f; // Influence at second keyframe (%)
67
68 // Display values (from integration)
69 float velocityAtStart = 0.0f; // Velocity at curve start
70 float velocityAtEnd = 0.0f; // Velocity at curve end
71 float accelerationMax = 0.0f; // Maximum acceleration
72
73 // Preset slot for save/load
74 int presetSlot = -1; // -1 if not using preset slot
75};
76
77// Keyframe module settings
78struct KeyframeSettings {
79 int slotCount = 4; // Number of preset slots
80 bool showVelocityGraph = true; // Show velocity visualization
81 bool showNumericValues = true; // Show speed/influence numbers
82};
