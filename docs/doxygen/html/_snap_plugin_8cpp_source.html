67/*****************************************************************************
68 * IsTextInputFocused
69 * Check if focus is on a text input field (Edit, RichEdit, etc.)
70 * Returns true if user is typing in a text field
71 *****************************************************************************/
72#ifdef MSWindows
73bool IsTextInputFocused() {
74 HWND fg = GetForegroundWindow();
75 if (!fg) return false;
76
77 DWORD threadId = GetWindowThreadProcessId(fg, NULL);
78 GUITHREADINFO gti = {0};
79 gti.cbSize = sizeof(GUITHREADINFO);
80
81 if (!GetGUIThreadInfo(threadId, &gti)) return false;
82
83 // Method 1: Check if caret is visible (most reliable for text input)
84 // hwndCaret is set when a text caret (cursor) is visible in any window
85 if (gti.hwndCaret != NULL) {
86 return true;
87 }
88
89 // Method 2: Check if caret is blinking (alternative detection)
90 if (gti.flags & GUI_CARETBLINKING) {
91 return true;
92 }
93
94 // Method 3: Check focused window class name (fallback)
95 HWND focused = gti.hwndFocus;
96 if (!focused) return false;
97
98 char className[64] = {0};
99 GetClassNameA(focused, className, sizeof(className));
100
101 // Common text input class names
102 if (_strnicmp(className, "Edit", 4) == 0 ||
103 _strnicmp(className, "RichEdit", 8) == 0 ||
104 _strnicmp(className, "RICHEDIT", 8) == 0 ||
105 _strnicmp(className, "Scintilla", 9) == 0) {
106 return true;
107 }
108
109 return false;
110}

333/*****************************************************************************
334 * ExecuteScript
335 * Execute ExtendScript using AEGP_ExecuteScript, returns result string
336 *****************************************************************************/
337A_Err ExecuteScript(const char *script, char *resultBuf = NULL,
338 size_t bufSize = 0) {
339 A_Err err = A_Err_NONE;
340
341 try {
342 AEGP_SuiteHandler suites(g_globals.pica_basicP);
343
344 AEGP_MemHandle resultH = NULL;
345 AEGP_MemHandle errorH = NULL;
346
347 err = suites.UtilitySuite6()->AEGP_ExecuteScript(
348 g_globals.plugin_id, script, TRUE, &resultH, &errorH);
349
350 // Get result if requested
351 if (resultBuf && bufSize > 0 && resultH) {
352 A_char *resultStr = NULL;
353 suites.MemorySuite1()->AEGP_LockMemHandle(resultH, (void **)&resultStr);
354 if (resultStr) {
355 strncpy(resultBuf, resultStr, bufSize - 1);
356 resultBuf[bufSize - 1] = '\0';
357 }
358 suites.MemorySuite1()->AEGP_UnlockMemHandle(resultH);
359 }
360
361 if (resultH) {
362 suites.MemorySuite1()->AEGP_FreeMemHandle(resultH);
363 }
364 if (errorH) {
365 suites.MemorySuite1()->AEGP_FreeMemHandle(errorH);
366 }
367
368 } catch (...) {
369 err = A_Err_GENERIC;
370 }
371
372 return err;
373}

416/*****************************************************************************
417 * LoadSettingsFromFile
418 * Read settings from CEP's settings file (cross-platform)
419 *****************************************************************************/
420void LoadSettingsFromFile() {
421#ifdef MSWindows
422 // Windows: %APPDATA%\Adobe\CEP\extensions\com.anchor.snap\settings.json
423 char path[512];
424 const char *appdata = getenv("APPDATA");
425 if (!appdata)
426 return;
427 snprintf(path, sizeof(path),
428 "%s\\Adobe\\CEP\\extensions\\com.anchor.snap\\settings.json",
429 appdata);
430#else
431 // macOS: ~/Library/Application
432 // Support/Adobe/CEP/extensions/com.anchor.snap/settings.json
433 char path[512];
434 const char *home = getenv("HOME");
435 if (!home)
436 return;
437 snprintf(path, sizeof(path),
438 "%s/Library/Application "
439 "Support/Adobe/CEP/extensions/com.anchor.snap/settings.json",
440 home);
441#endif
442
443 FILE *f = fopen(path, "r");
444 if (!f)
445 return;
446
447 char buffer[2048];
448 size_t len = fread(buffer, 1, sizeof(buffer) - 1, f);
449 buffer[len] = '\0';
450 fclose(f);
451
452 // Parse simple JSON
453 NativeUI::GridSettings &settings = NativeUI::GetSettings();
454
455 const char *p;
456 // useCompMode
457 if ((p = strstr(buffer, "\"useCompMode\":")) != NULL) {
458 p += 14; // skip past key
459 settings.useCompMode =
460 (strstr(p, "true") != NULL &&
461 (strstr(p, "true") < strstr(p, ",") || strstr(p, ",") == NULL));
462 }
463 // useMaskRecognition
464 if ((p = strstr(buffer, "\"useMaskRecognition\":")) != NULL) {
465 p += 20; // "useMaskRecognition": = 20 chars
466 settings.useMaskRecognition =
467 (strstr(p, "true") != NULL &&
468 (strstr(p, "true") < strstr(p, ",") || strstr(p, ",") == NULL));
469 }
470 // settingsPanelOpen
471 if ((p = strstr(buffer, "\"settingsPanelOpen\":")) != NULL) {
472 p += 20; // "settingsPanelOpen": = 20 chars
473 settings.settingsPanelOpen =
474 (strstr(p, "true") != NULL &&
475 (strstr(p, "true") < strstr(p, ",") || strstr(p, ",") == NULL));
476 }
477 // gridWidth
478 if ((p = strstr(buffer, "\"gridWidth\":")) != NULL) {
479 p += 12;
480 int val = atoi(p);
481 if (val >= 3 && val <= 7) {
482 g_loadedGridWidth = val;
483 }
484 }
485 // gridHeight
486 if ((p = strstr(buffer, "\"gridHeight\":")) != NULL) {
487 p += 13;
488 int val = atoi(p);
489 if (val >= 3 && val <= 7) {
490 g_loadedGridHeight = val;
491 }
492 }
493 // gridScale (0-9)
494 if ((p = strstr(buffer, "\"gridScale\":")) != NULL) {
495 p += 12;
496 int val = atoi(p);
497 if (val >= 0 && val <= 9) {
498 g_loadedGridScale = val;
499 }
500 }
501 // gridOpacity
502 if ((p = strstr(buffer, "\"gridOpacity\":")) != NULL) {
503 p += 14;
504 int val = atoi(p);
505 if (val >= 0 && val <= 100) {
506 g_loadedGridOpacity = val;
507 }
508 }
509 // cellOpacity
510 if ((p = strstr(buffer, "\"cellOpacity\":")) != NULL) {
511 p += 14;
512 int val = atoi(p);
513 if (val >= 0 && val <= 100) {
514 g_loadedCellOpacity = val;
515 }
516 }
517
518 // customAnchors array - parse [{"x":50,"y":50},{"x":50,"y":50},{"x":50,"y":50}]
519 if ((p = strstr(buffer, "\"customAnchors\":")) != NULL) {
520 // Find the array start
521 p = strchr(p, '[');
522 if (p) {
523 NativeUI::GridSettings &settings = NativeUI::GetSettings();
524 // Parse each anchor object
525 for (int i = 0; i < 3; i++) {
526 // Find "x":
527 const char *xp = strstr(p, "\"x\":");
528 if (!xp)
529 break;
530 xp += 4;
531 // Skip whitespace
532 while (*xp == ' ' || *xp == '\t')
533 xp++;
534 int xVal = atoi(xp);
535
536 // Find "y":
537 const char *yp = strstr(xp, "\"y\":");
538 if (!yp)
539 break;
540 yp += 4;
541 // Skip whitespace
542 while (*yp == ' ' || *yp == '\t')
543 yp++;
544 int yVal = atoi(yp);
545
546 // Convert from 0-100 percent to 0.0-1.0 ratio
547 float rx = xVal / 100.0f;
548 float ry = yVal / 100.0f;
549
550 switch (i) {
551 case 0:
552 settings.customAnchor1X = rx;
553 settings.customAnchor1Y = ry;
554 break;
555 case 1:
556 settings.customAnchor2X = rx;
557 settings.customAnchor2Y = ry;
558 break;
559 case 2:
560 settings.customAnchor3X = rx;
561 settings.customAnchor3Y = ry;
562 break;
563 }
564
565 // Move to next object
566 p = strchr(yp, '}');
567 if (!p)
568 break;
569 }
570 }
571 }
572
573 // clipboardAnchor - parse {"x":0.5,"y":0.5} (ratio 0-1)
574 if ((p = strstr(buffer, "\"clipboardAnchor\":")) != NULL) {
575 // Check if it's not null
576 const char *nullCheck = strstr(p, "null");
577 const char *objStart = strchr(p, '{');
578 if (objStart && (!nullCheck || objStart < nullCheck)) {
579 const char *xp = strstr(objStart, "\"x\":");
580 if (xp) {
581 xp += 4;
582 while (*xp == ' ' || *xp == '\t')
583 xp++;
584 float rx = (float)atof(xp);
585
586 const char *yp = strstr(xp, "\"y\":");
587 if (yp) {
588 yp += 4;
589 while (*yp == ' ' || *yp == '\t')
590 yp++;
591 float ry = (float)atof(yp);
592
593 // Store in NativeUI clipboard
594 NativeUI::SetClipboardAnchor(rx, ry);
595 }
596 }
597 }
598 }
599}

896/*****************************************************************************
897 * ApplyAnchorToLayers
898 * Apply anchor point to selected layers based on grid position
899 * Respects composition mode and mask recognition settings
900 *****************************************************************************/
901void ApplyAnchorToLayers(int gridX, int gridY) {
902 int gridW = g_loadedGridWidth;
903 int gridH = g_loadedGridHeight;
904
905 // Get current mode settings
906 NativeUI::GridSettings &settings = NativeUI::GetSettings();
907 bool useCompMode = settings.useCompMode;
908 bool useMaskMode = settings.useMaskRecognition;
909
910 char script[6000];
911 snprintf(
912 script, sizeof(script),
913 "(function(){"
914 "var gx=%d,gy=%d,gridW=%d,gridH=%d;"
915 "var useCompMode=%s,useMaskMode=%s;"
916 "var c=app.project.activeItem;"
917 "if(!c||!(c instanceof CompItem))return;"
918 "if(c.selectedLayers.length==0)return;"
919 "app.beginUndoGroup('Set Anchor');"
920 "for(var i=0;i<c.selectedLayers.length;i++){"
921 "var L=c.selectedLayers[i];"
922 "var b=null;"
923 // Composition mode: use comp bounds
924 "if(useCompMode){"
925 "b={left:0,top:0,width:c.width,height:c.height};"
926 "}else{"
927 // Selection mode: try mask first if enabled, then sourceRect
928 "if(useMaskMode){"
929 "var masks=L.property('ADBE Mask Parade');"
930 "if(masks&&masks.numProperties>0){"
931 "var minX=Infinity,minY=Infinity,maxX=-Infinity,maxY=-Infinity;"
932 "for(var m=1;m<=masks.numProperties;m++){"
933 "var mask=masks.property(m);"
934 "var path=mask.property('ADBE Mask Shape').valueAtTime(c.time,false);"
935 "if(!path||!path.vertices)continue;"
936 "var verts=path.vertices;"
937 "for(var v=0;v<verts.length;v++){"
938 "var pt=verts[v];"
939 "if(pt[0]<minX)minX=pt[0];"
940 "if(pt[0]>maxX)maxX=pt[0];"
941 "if(pt[1]<minY)minY=pt[1];"
942 "if(pt[1]>maxY)maxY=pt[1];"
943 "}}"
944 "if(minX!=Infinity){"
945 "b={left:minX,top:minY,width:maxX-minX,height:maxY-minY};"
946 "}}}"
947 // Fall back to sourceRectAtTime
948 "if(!b)b=L.sourceRectAtTime(c.time,false);"
949 "}"
950 "if(!b||b.width<=0||b.height<=0)continue;"
951 "var px=gx/(gridW-1),py=gy/(gridH-1);"
952 // Calculate anchor in appropriate coordinate space
953 "var nx,ny;"
954 "if(useCompMode){"
955 // Composition mode: convert comp coord to layer local
956 // Get layer transform properties
957 "var pos=L.position.value;"
958 "var anc=L.anchorPoint.value;"
959 "var sc=L.scale.value;"
960 "var rot=L.rotation.value*Math.PI/180;"
961 // Target position in comp space
962 "var compX=c.width*px,compY=c.height*py;"
963 // Translate: comp coord relative to layer position
964 "var dx=compX-pos[0],dy=compY-pos[1];"
965 // Reverse rotation
966 "var cos_r=Math.cos(-rot),sin_r=Math.sin(-rot);"
967 "var rx=dx*cos_r-dy*sin_r,ry=dx*sin_r+dy*cos_r;"
968 // Reverse scale
969 "var sx=100/sc[0],sy=100/sc[1];"
970 "rx=rx*sx;ry=ry*sy;"
971 // Add anchor offset to get local coord
972 "nx=rx+anc[0];ny=ry+anc[1];"
973 "}else{"
974 // Selection mode: already in layer local coordinates
975 "nx=b.left+b.width*px;ny=b.top+b.height*py;"
976 "}"
977 "var ap=L.property('ADBE Transform Group').property('ADBE Anchor "
978 "Point');"
979 "var pp=L.property('ADBE Transform Group').property('ADBE Position');"
980 "if(!ap||!pp)continue;"
981 "var oa=ap.value,pos=pp.value;"
982 "var dx=nx-oa[0],dy=ny-oa[1];"
983 "var sc=L.property('ADBE Transform Group').property('ADBE "
984 "Scale').value;"
985 "var sx=sc[0]/100,sy=sc[1]/100;"
986 "var rot=L.property('ADBE Transform Group').property('ADBE Rotate "
987 "Z').value*Math.PI/180;"
988 "var rdx=(dx*Math.cos(rot)-dy*Math.sin(rot))*sx;"
989 "var rdy=(dx*Math.sin(rot)+dy*Math.cos(rot))*sy;"
990 "ap.setValue([nx,ny]);"
991 "if(pos.length==3)pp.setValue([pos[0]+rdx,pos[1]+rdy,pos[2]]);"
992 "else pp.setValue([pos[0]+rdx,pos[1]+rdy]);"
993 "}"
994 "app.endUndoGroup();"
995 "})();",
996 gridX, gridY, gridW, gridH, useCompMode ? "true" : "false",
997 useMaskMode ? "true" : "false");
998
999 ExecuteScript(script);
1000}
