140/*****************************************************************************
141 * Initialize

143void Initialize() {
144 if (g_hwnd) return;
145
146 // Initialize GDI+
147 GdiplusStartupInput gdiplusStartupInput;
148 Status gdipStatus = GdiplusStartup(&g_gdiplusToken, &gdiplusStartupInput, NULL);
149 if (gdipStatus != Ok) {
150 // GDI+ initialization failed - cannot proceed
151 return;
152 }
153
154 // Register main window class (explicitly use Wide version for Unicode strings)
155 WNDCLASSEXW wc = {0};
156 wc.cbSize = sizeof(WNDCLASSEXW);
157 wc.style = CS_HREDRAW | CS_VREDRAW | CS_DBLCLKS;
158 wc.lpfnWndProc = WndProc;
159 wc.hInstance = GetModuleHandle(NULL);
160 wc.hCursor = LoadCursor(NULL, IDC_ARROW);
161 wc.hbrBackground = NULL;
162 wc.lpszClassName = L"TextUIWindow";
163 ATOM classAtom = RegisterClassExW(&wc);
164 if (classAtom == 0 && GetLastError() != ERROR_CLASS_ALREADY_EXISTS) {
165 // Registration failed and class doesn't exist
166 GdiplusShutdown(g_gdiplusToken);
167 g_gdiplusToken = 0;
168 return;
169 }
170
171 // Create main window (initially hidden)
172 g_hwnd = CreateWindowExW(
173 WS_EX_TOPMOST | WS_EX_TOOLWINDOW | WS_EX_NOACTIVATE,
174 L"TextUIWindow",
175 L"Text Options",
176 WS_POPUP,
177 0, 0, WINDOW_WIDTH, WINDOW_HEIGHT,
178 NULL, NULL, GetModuleHandle(NULL), NULL
179 );
180
181 if (!g_hwnd) {
182 // Window creation failed
183 GdiplusShutdown(g_gdiplusToken);
184 g_gdiplusToken = 0;
185 return;
186 }
187
188 SetLayeredWindowAttributes(g_hwnd, 0, 255, LWA_ALPHA);
189
190 // Register color picker window class (non-critical, ignore failure)
191 WNDCLASSEXW wcPicker = {0};
192 wcPicker.cbSize = sizeof(WNDCLASSEXW);
193 wcPicker.style = CS_HREDRAW | CS_VREDRAW;
194 wcPicker.lpfnWndProc = ColorPickerWndProc;
195 wcPicker.hInstance = GetModuleHandle(NULL);
196 wcPicker.hCursor = LoadCursor(NULL, IDC_ARROW);
197 wcPicker.hbrBackground = NULL;
198 wcPicker.lpszClassName = L"TextUIColorPicker";
199 RegisterClassExW(&wcPicker);
200}

220/*****************************************************************************
221 * ShowPanel
222 *****************************************************************************/
223void ShowPanel(int x, int y) {
224 if (!g_hwnd) Initialize();
225
226 // Reset state
227 g_result = TextResult();
228 g_dragging = false;
229 g_editMode = false;
230 g_hoverTarget = TARGET_NONE;
231
232 // Position window
233 RECT workArea;
234 SystemParametersInfo(SPI_GETWORKAREA, 0, &workArea, 0);
235
236 int posX = x - WINDOW_WIDTH / 2;
237 int posY = y - WINDOW_HEIGHT / 2;
238
239 if (posX < workArea.left) posX = workArea.left;
240 if (posY < workArea.top) posY = workArea.top;
241 if (posX + WINDOW_WIDTH > workArea.right) posX = workArea.right - WINDOW_WIDTH;
242 if (posY + WINDOW_HEIGHT > workArea.bottom) posY = workArea.bottom - WINDOW_HEIGHT;
243
244 SetWindowPos(g_hwnd, HWND_TOPMOST, posX, posY, WINDOW_WIDTH, WINDOW_HEIGHT,
245 SWP_NOACTIVATE | SWP_SHOWWINDOW);
246 ShowWindow(g_hwnd, SW_SHOWNA);
247 InvalidateRect(g_hwnd, NULL, TRUE);
248 g_visible = true;
249}

269/*****************************************************************************
270 * UpdateHover
271 *****************************************************************************/
272void UpdateHover(int mouseX, int mouseY) {
273 if (!g_hwnd || !g_visible || g_dragging) return;
274
275 RECT wndRect;
276 GetWindowRect(g_hwnd, &wndRect);
277 int localX = mouseX - wndRect.left;
278 int localY = mouseY - wndRect.top;
279
280 bool needsRepaint = false;
281 POINT pt = {localX, localY};
282
283 // Check value boxes
284 ValueTarget newHover = HitTestValue(localX, localY);
285 if (newHover != g_hoverTarget) {
286 g_hoverTarget = newHover;
287 needsRepaint = true;
288
289 // Set cursor
290 if (g_hoverTarget != TARGET_NONE && !g_editMode) {
291 SetCursor(LoadCursor(NULL, IDC_SIZEWE));
292 } else {
293 SetCursor(LoadCursor(NULL, IDC_ARROW));
294 }
295 }
296
297 // Check align buttons
298 int newAlign = HitTestAlign(localX, localY);
299 if (newAlign != g_hoverAlign) {
300 g_hoverAlign = newAlign;
301 needsRepaint = true;
302 }
303
304 // Check color boxes
305 bool newFillHover = PtInRect(&g_fillColorRect, pt);
306 bool newStrokeHover = PtInRect(&g_strokeColorRect, pt);
307 if (newFillHover != g_fillColorHover || newStrokeHover != g_strokeColorHover) {
308 g_fillColorHover = newFillHover;
309 g_strokeColorHover = newStrokeHover;
310 needsRepaint = true;
311 }
312
313 // Check pin/close
314 bool newPinHover = PtInRect(&g_pinRect, pt);
315 bool newCloseHover = PtInRect(&g_closeRect, pt);
316 if (newPinHover != g_pinHover || newCloseHover != g_closeHover) {
317 g_pinHover = newPinHover;
318 g_closeHover = newCloseHover;
319 needsRepaint = true;
320 }
321
322 if (needsRepaint) {
323 InvalidateRect(g_hwnd, NULL, FALSE);
324 }
325}

330 * SetTextInfo - Parse JSON from ExtendScript
331 *****************************************************************************/
332void SetTextInfo(const wchar_t* jsonInfo) {
333 if (!jsonInfo || jsonInfo[0] == L'\0') return;
334
335 // Simple JSON parsing (assuming well-formed)
336 std::wstring json(jsonInfo);
337
338 auto getFloat = [&json](const wchar_t* key) -> float {
339 std::wstring search = std::wstring(L"\"") + key + L"\":";
340 size_t pos = json.find(search);
341 if (pos == std::wstring::npos) return 0;
342 pos += search.length();
343 return (float)_wtof(json.c_str() + pos);
344 };
345
346 auto getBool = [&json](const wchar_t* key) -> bool {
347 std::wstring search = std::wstring(L"\"") + key + L"\":";
348 size_t pos = json.find(search);
349 if (pos == std::wstring::npos) return false;
350 pos += search.length();
351 while (json[pos] == L' ') pos++;
352 return json[pos] == L't';
353 };
354
355 auto getString = [&json](const wchar_t* key, wchar_t* out, int maxLen) {
356 std::wstring search = std::wstring(L"\"") + key + L"\":\"";
357 size_t pos = json.find(search);
358 if (pos == std::wstring::npos) { out[0] = L'\0'; return; }
359 pos += search.length();
360 size_t end = json.find(L'"', pos);
361 if (end == std::wstring::npos) { out[0] = L'\0'; return; }
362 int len = (int)(end - pos);
363 if (len >= maxLen) len = maxLen - 1;
364 wcsncpy(out, json.c_str() + pos, len);
365 out[len] = L'\0';
366 };
367
368 getString(L"font", g_textInfo.font, 128);
369 getString(L"layerName", g_textInfo.layerName, 256);
370 g_textInfo.fontSize = getFloat(L"fontSize");
371 g_textInfo.tracking = getFloat(L"tracking");
372 g_textInfo.leading = getFloat(L"leading");
373 g_textInfo.strokeWidth = getFloat(L"strokeWidth");
374 g_textInfo.applyFill = getBool(L"applyFill");
375 g_textInfo.applyStroke = getBool(L"applyStroke");
376 g_textInfo.justify = (Justification)(int)getFloat(L"justification");
377

379 // TODO: proper array parsing
380
381 InvalidateRect(g_hwnd, NULL, FALSE);
382}

384/*****************************************************************************
385 * WndProc
386 *****************************************************************************/
387static LRESULT CALLBACK WndProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam) {
388 switch (msg) {
389 case WM_PAINT: {
390 PAINTSTRUCT ps;
391 HDC hdc = BeginPaint(hwnd, &ps);
392 Draw(hdc);
393 EndPaint(hwnd, &ps);
394 return 0;
395 }
396
397 case WM_MOUSEACTIVATE:
398 // Prevent activation when clicked (unless in edit mode)
399 if (!g_editMode) return MA_NOACTIVATE;
400 return MA_ACTIVATE;
401
402 case WM_LBUTTONDOWN:
403 HandleMouseDown(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
404 return 0;
405
406 case WM_LBUTTONUP:
407 HandleMouseUp(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
408 return 0;
409
410 case WM_MOUSEMOVE:
411 HandleMouseMove(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
412 return 0;
413
414 case WM_LBUTTONDBLCLK:
415 HandleDoubleClick(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
416 return 0;
417
418 case WM_KEYDOWN:
419 HandleKeyDown(wParam);
420 return 0;
421
422 case WM_CHAR:
423 HandleChar((wchar_t)wParam);
424 return 0;
425
426 case WM_ACTIVATE:
427 if (LOWORD(wParam) == WA_INACTIVE && !g_keepPanelOpen && !g_dragging && !g_editMode) {
428 g_result.cancelled = true;
429 ShowWindow(hwnd, SW_HIDE);
430 g_visible = false;
431 }
432 return 0;
433
434 case WM_KILLFOCUS:
435 if (g_editMode) {
436 ExitEditMode(true);
437 } else if (!g_keepPanelOpen && !g_dragging) {
438 g_result.cancelled = true;
439 ShowWindow(hwnd, SW_HIDE);
440 g_visible = false;
441 }
442 return 0;
443
444 case WM_SETCURSOR:
445 if (g_hoverTarget != TARGET_NONE && !g_editMode) {
446 SetCursor(LoadCursor(NULL, IDC_SIZEWE));
447 return TRUE;
448 }
449 break;
450
451 case WM_DESTROY:
452 g_visible = false;
453 return 0;
454 }
455
456 return DefWindowProc(hwnd, msg, wParam, lParam);
457}

485/*****************************************************************************
486 * DrawHeader
487 *****************************************************************************/
488static void DrawHeader(Graphics& g) {
489 SolidBrush headerBrush(COLOR_HEADER_BG);
490 g.FillRectangle(&headerBrush, 0, 0, WINDOW_WIDTH, HEADER_HEIGHT);
491
492 FontFamily fontFamily(L"Segoe UI");
493 Font titleFont(&fontFamily, 11, FontStyleBold, UnitPixel);
494 Font layerFont(&fontFamily, 10, FontStyleRegular, UnitPixel);
495 SolidBrush textBrush(COLOR_TEXT);
496 SolidBrush dimBrush(COLOR_TEXT_DIM);
497
498 // Title
499 g.DrawString(L"Text Options", -1, &titleFont, PointF(10, 8), &textBrush);
500
501 // Layer name
502 if (g_textInfo.layerName[0] != L'\0') {
503 RectF layerRect(120, 8, 140, 20);
504 StringFormat sf;
505 sf.SetAlignment(StringAlignmentFar);
506 sf.SetTrimming(StringTrimmingEllipsisCharacter);
507 g.DrawString(g_textInfo.layerName, -1, &layerFont, layerRect, &sf, &dimBrush);
508 }
509
510 // Pin button
511 int btnY = (HEADER_HEIGHT - 20) / 2;
512 g_pinRect = {WINDOW_WIDTH - 54, btnY, WINDOW_WIDTH - 34, btnY + 20};
513 Color pinColor = g_keepPanelOpen ? COLOR_PIN_ACTIVE :
514 (g_pinHover ? COLOR_ALIGN_HOVER : COLOR_ALIGN_BG);
515 SolidBrush pinBrush(pinColor);
516 g.FillRectangle(&pinBrush, g_pinRect.left, g_pinRect.top, 20, 20);
517
518 Pen pinPen(g_keepPanelOpen ? Color(255, 40, 40, 40) : COLOR_TEXT, 1.5f);
519 int px = g_pinRect.left + 10, py = g_pinRect.top + 10;
520 g.DrawLine(&pinPen, px - 4, py - 4, px + 4, py + 4);
521 g.DrawEllipse(&pinPen, px - 3, py - 6, 6, 6);
522
523 // Close button
524 g_closeRect = {WINDOW_WIDTH - 28, btnY, WINDOW_WIDTH - 8, btnY + 20};
525 Color closeColor = g_closeHover ? COLOR_CLOSE_HOVER : COLOR_ALIGN_BG;
526 SolidBrush closeBrush(closeColor);
527 g.FillRectangle(&closeBrush, g_closeRect.left, g_closeRect.top, 20, 20);
528
529 Pen closePen(COLOR_TEXT, 1.5f);
530 int cx = g_closeRect.left + 10, cy = g_closeRect.top + 10;
531 g.DrawLine(&closePen, cx - 4, cy - 4, cx + 4, cy + 4);
532 g.DrawLine(&closePen, cx + 4, cy - 4, cx - 4, cy + 4);
533}
