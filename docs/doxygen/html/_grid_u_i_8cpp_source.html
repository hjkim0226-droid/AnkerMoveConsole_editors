100bool Initialize() {
101 if (g_initialized)
102 return true;
103
104 // Initialize GDI+
105 Gdiplus::GdiplusStartupInput gdiplusStartupInput;
106 if (Gdiplus::GdiplusStartup(&g_gdiplusToken, &gdiplusStartupInput, NULL) !=
107 Gdiplus::Ok) {
108 return false;
109 }
110
111 g_hInstance = GetModuleHandle(NULL);
112
113 // Register window class
114 WNDCLASSEXW wc = {0};
115 wc.cbSize = sizeof(WNDCLASSEXW);
116 wc.style = CS_HREDRAW | CS_VREDRAW;
117 wc.lpfnWndProc = GridWndProc;
118 wc.hInstance = g_hInstance;
119 wc.hCursor = LoadCursor(NULL, IDC_ARROW);
120 wc.hbrBackground = CreateSolidBrush(COLOR_BG);
121 wc.lpszClassName = GRID_CLASS_NAME;
122
123 if (!RegisterClassExW(&wc)) {
124 Gdiplus::GdiplusShutdown(g_gdiplusToken);
125 return false;
126 }
127
128 g_initialized = true;
129 return true;
130}

132void Cleanup() {
133 if (g_gridWnd) {
134 DestroyWindow(g_gridWnd);
135 g_gridWnd = NULL;
136 }
137 if (g_initialized) {
138 UnregisterClassW(GRID_CLASS_NAME, g_hInstance);
139 g_initialized = false;
140 }
141 if (g_gdiplusToken != 0) {
142 Gdiplus::GdiplusShutdown(g_gdiplusToken);
143 g_gdiplusToken = 0;
144 }

147void ShowGrid(int mouseX, int mouseY, const GridConfig &config) {
148 if (!g_initialized && !Initialize())
149 return;
150
151 g_config = config;
152 g_hoverCellX = -1;
153 g_hoverCellY = -1;
154 g_hoverExtOption = OPT_NONE;
155
156 // FIXED grid area size with scale applied
157 // 10 steps: -20% to +70% (scale factors 0.8 to 1.7)
158 float scaleFactors[] = {0.8f, 0.9f, 1.0f, 1.1f, 1.2f, 1.3f, 1.4f, 1.5f, 1.6f, 1.7f};
159 int scaleIndex = 2; // Default to 0%
160 // Use config.cellSize to infer scale (40 base * factor)
161 if (g_config.cellSize <= 32)
162 scaleIndex = 0; // 40 * 0.8 = 32
163 else if (g_config.cellSize <= 36)
164 scaleIndex = 1; // 40 * 0.9 = 36
165 else if (g_config.cellSize <= 40)
166 scaleIndex = 2; // 40 * 1.0 = 40
167 else if (g_config.cellSize <= 44)
168 scaleIndex = 3; // 40 * 1.1 = 44
169 else if (g_config.cellSize <= 48)
170 scaleIndex = 4; // 40 * 1.2 = 48
171 else if (g_config.cellSize <= 52)
172 scaleIndex = 5; // 40 * 1.3 = 52
173 else if (g_config.cellSize <= 56)
174 scaleIndex = 6; // 40 * 1.4 = 56
175 else if (g_config.cellSize <= 60)
176 scaleIndex = 7; // 40 * 1.5 = 60
177 else if (g_config.cellSize <= 64)
178 scaleIndex = 8; // 40 * 1.6 = 64
179 else
180 scaleIndex = 9; // 40 * 1.7 = 68
181
182 int baseGridSize = 120; // Fixed base size for grid area
183 g_fixedGridPixels = (int)(baseGridSize * scaleFactors[scaleIndex]);
184
185 // Update scaled icon dimensions based on scale factor
186 g_currentScale = scaleFactors[scaleIndex];
187 g_sidePanelWidth = (int)(BASE_SIDE_PANEL_WIDTH * g_currentScale);
188 g_iconSize = (int)(BASE_ICON_SIZE * g_currentScale);
189 g_iconSpacing = (int)(BASE_ICON_SPACING * g_currentScale);
190
191 // Scale margin (base margin = 2)
192 g_scaledMargin = (int)(g_config.margin * g_currentScale);
193 if (g_scaledMargin < 1) g_scaledMargin = 1;
194
195 // Calculate cellSize based on max dimension (like FX Console null parents)
196 int maxDim = (g_config.gridWidth > g_config.gridHeight) ? g_config.gridWidth
197 : g_config.gridHeight;
198 int spacing = g_config.spacing;
199 int calculatedCellSize =
200 (g_fixedGridPixels - (maxDim - 1) * spacing) / maxDim;
201 if (calculatedCellSize < 10)
202 calculatedCellSize = 10; // Minimum cell size
203
204 // Override cellSize with calculated value
205 g_config.cellSize = calculatedCellSize;
206 int cellTotal = g_config.cellSize + spacing;
207
208 // Actual grid pixels (always fits within g_fixedGridPixels)
209 int gridPixelsW = g_config.gridWidth * cellTotal;
210 int gridPixelsH = g_config.gridHeight * cellTotal;
211
212 // Window: FIXED size (use scaled margin)
213 g_windowWidth = g_sidePanelWidth + g_fixedGridPixels + g_scaledMargin * 2 +
214 g_sidePanelWidth;
215
216 int minHeight =
217 g_iconSize * 3 + g_iconSpacing * 2 + (int)(20 * g_currentScale);
218 int bottomButtonsHeight = g_iconSize + (int)(10 * g_currentScale);
219 int gridAreaHeight = g_fixedGridPixels + g_scaledMargin * 2;
220 int baseHeight = (gridAreaHeight > minHeight) ? gridAreaHeight : minHeight;
221 g_windowHeight = baseHeight + bottomButtonsHeight;
222
223 // Calculate vertical padding to center grid area when window is taller due to icons
224 g_gridVerticalPadding = (baseHeight - gridAreaHeight) / 2;
225
226 // Grid offset: center the actual grid within the fixed area
227 g_gridOffsetX = (g_fixedGridPixels - gridPixelsW) / 2;
228 g_gridOffsetY = (g_fixedGridPixels - gridPixelsH) / 2;
229
230 // Mouse at WINDOW CENTER (grid area center, accounting for vertical padding)
231 int gridCenterX = g_sidePanelWidth + g_scaledMargin + g_fixedGridPixels / 2;
232 int gridCenterY = g_gridVerticalPadding + g_scaledMargin + g_fixedGridPixels / 2;
233 g_windowX = mouseX - gridCenterX;
234 g_windowY = mouseY - gridCenterY;
235
236 // Apply grid opacity setting (0-100 -> 0-255)
237 BYTE alpha = (BYTE)(g_settings.gridOpacity * 255 / 100);
238 if (alpha < 100)
239 alpha = 100; // Minimum visibility
240
241 if (g_gridWnd) {
242 SetWindowPos(g_gridWnd, HWND_TOPMOST, g_windowX, g_windowY, g_windowWidth,
243 g_windowHeight, SWP_SHOWWINDOW);
244 SetLayeredWindowAttributes(g_gridWnd, COLOR_BG, alpha,
245 LWA_COLORKEY | LWA_ALPHA);
246 } else {
247 g_gridWnd = CreateWindowExW(
248 WS_EX_TOOLWINDOW | WS_EX_TOPMOST | WS_EX_LAYERED, GRID_CLASS_NAME, NULL,
249 WS_POPUP | WS_VISIBLE, g_windowX, g_windowY, g_windowWidth,
250 g_windowHeight, NULL, NULL, g_hInstance, NULL);
251
252 SetLayeredWindowAttributes(g_gridWnd, COLOR_BG, alpha,
253 LWA_COLORKEY | LWA_ALPHA);
254 }
255
256 UpdateHoverFromMouse(mouseX, mouseY);
257 InvalidateRect(g_gridWnd, NULL, FALSE); // FALSE to prevent flickering
258}

260GridResult HideGrid(int mouseX, int mouseY) {
261 GridResult result;
262
263 if (g_gridWnd) {
264 UpdateHoverFromMouse(mouseX, mouseY);
265 result.gridX = g_hoverCellX;
266 result.gridY = g_hoverCellY;
267 result.extendedOption = g_hoverExtOption;
268 result.cancelled =
269 (g_hoverCellX < 0 && g_hoverCellY < 0 && g_hoverExtOption == OPT_NONE);
270
271 ShowWindow(g_gridWnd, SW_HIDE);
272 } else {
273 result.cancelled = true;
274 }
275
276 return result;
277}

329// Calculate hover cell or extended option from screen coordinates
330static void UpdateHoverFromMouse(int screenX, int screenY) {
331 int cellTotal = g_config.cellSize + g_config.spacing;
332 int gridPixelsW = g_config.gridWidth * cellTotal;
333 int gridPixelsH = g_config.gridHeight * cellTotal;
334 // Apply grid offset for centered rectangular grids (with vertical padding)
335 int gridStartX = g_sidePanelWidth + g_scaledMargin + g_gridOffsetX;
336 int gridStartY = g_gridVerticalPadding + g_scaledMargin + g_gridOffsetY;
337
338 int relX = screenX - g_windowX;
339 int relY = screenY - g_windowY;
340
341 g_hoverCellX = -1;
342 g_hoverCellY = -1;
343 g_hoverExtOption = NativeUI::OPT_NONE;
344
345 // Check left panel (Custom Anchors only)
346 if (relX < g_sidePanelWidth && relX >= 0) {
347 int bottomButtonsHeight = g_iconSize + (int)(10 * g_currentScale);
348 int gridAreaHeight = g_windowHeight - bottomButtonsHeight;
349 int iconY = (gridAreaHeight - (g_iconSize * 3 + g_iconSpacing * 2)) / 2;
350
351 // Check custom anchor icons
352 for (int i = 0; i < 3; i++) {
353 int top = iconY + i * (g_iconSize + g_iconSpacing);
354 if (relY >= top && relY < top + g_iconSize) {
355 g_hoverExtOption =
356 (NativeUI::ExtendedOption)(NativeUI::OPT_CUSTOM_1 + i);
357 return;
358 }
359 }
360 return;
361 }
362
363 // Check right panel (Mode controls) - limit to panel width only
364 if (relX >= g_windowWidth - g_sidePanelWidth && relX < g_windowWidth) {
365 int bottomButtonsHeight = g_iconSize + (int)(10 * g_currentScale);
366 int gridAreaHeight = g_windowHeight - bottomButtonsHeight;
367 int iconY = (gridAreaHeight - (g_iconSize * 3 + g_iconSpacing * 2)) / 2;
368 for (int i = 0; i < 3; i++) {
369 int top = iconY + i * (g_iconSize + g_iconSpacing);
370 if (relY >= top && relY < top + g_iconSize) {
371 switch (i) {
372 case 0:
373 g_hoverExtOption = NativeUI::OPT_COMP_MODE;
374 break;
375 case 1:
376 g_hoverExtOption = NativeUI::OPT_MASK_MODE;
377 break;
378 case 2:
379 g_hoverExtOption = NativeUI::OPT_SETTINGS;
380 break;
381 }
382 return;
383 }
384 }
385 return;
386 }
387
388 // Check Copy/Paste buttons (below grid area, centered in window)
389 int bottomButtonsHeight = g_iconSize + (int)(10 * g_currentScale);
390 // Use global g_fixedGridPixels for button positioning (with scaled margin and vertical padding)
391 // Center buttons in the window
392 int windowCenterX =
393 g_sidePanelWidth + g_scaledMargin + g_fixedGridPixels / 2;
394 int gridBottomY = g_gridVerticalPadding + g_scaledMargin + g_fixedGridPixels + g_iconSize / 2 +
395 (int)(5 * g_currentScale);
396
397 if (relY >= gridBottomY - g_iconSize / 2 &&
398 relY < gridBottomY + g_iconSize / 2) {
399 int copyX = windowCenterX - g_iconSize / 2 - (int)(5 * g_currentScale);
400 int pasteX = windowCenterX + g_iconSize / 2 + (int)(5 * g_currentScale);
401
402 // Copy button
403 if (relX >= copyX - g_iconSize / 2 && relX < copyX + g_iconSize / 2) {
404 g_hoverExtOption = NativeUI::OPT_COPY_ANCHOR;
405 return;
406 }
407 // Paste button
408 if (relX >= pasteX - g_iconSize / 2 && relX < pasteX + g_iconSize / 2) {
409 g_hoverExtOption = NativeUI::OPT_PASTE_ANCHOR;
410 return;
411 }
412 }
413
414 // Check grid cells
415 int gridRelX = relX - gridStartX;
416 int gridRelY = relY - gridStartY;
417
418 if (gridRelX >= 0 && gridRelX < gridPixelsW && gridRelY >= 0 &&
419 gridRelY < gridPixelsH) {
420 int cellX = gridRelX / cellTotal;
421 int cellY = gridRelY / cellTotal;
422 if (cellX < g_config.gridWidth && cellY < g_config.gridHeight) {
423 g_hoverCellX = cellX;
424 g_hoverCellY = cellY;
425 }
426 }
427}

441// Draw an icon based on type using pure GDI+
442static void DrawIcon(HDC hdc, int cx, int cy, NativeUI::ExtendedOption type,
443 bool hover, bool active) {
444 using namespace Gdiplus;
445
446 // Draw hover background first
447 DrawIconBackground(hdc, cx, cy, hover);
448
449 // Create GDI+ Graphics with anti-aliasing
450 Graphics graphics(hdc);
451 graphics.SetSmoothingMode(SmoothingModeNone);
452 graphics.SetPixelOffsetMode(PixelOffsetModeHalf);
453 graphics.SetTextRenderingHint(TextRenderingHintClearTypeGridFit);
454
455 int r = g_iconSize / 2 - (int)(6 * g_currentScale);
456 int s = (int)(3 * g_currentScale);
457
458 // Helper to convert COLORREF to GDI+ Color
459 auto toColor = [](COLORREF c) {
460 return Color(255, GetRValue(c), GetGValue(c), GetBValue(c));
461 };
462
463 switch (type) {
464 case NativeUI::OPT_CUSTOM_1:
465 case NativeUI::OPT_CUSTOM_2:
466 case NativeUI::OPT_CUSTOM_3: {
467 // Default: dark gray, hover: blue (like selection mode)
468 COLORREF colorRef =
469 hover ? COLOR_ICON_HOVER : (active ? COLOR_BLUE : COLOR_DARK_GRAY);
470 Color color = toColor(colorRef);
471 // Use hover background color when hovering, otherwise transparent key
472 COLORREF bgColorRef = hover ? RGB(50, 60, 70) : COLOR_BG;
473 Color bgColor = toColor(bgColorRef);
474 Pen pen(color, 2.0f);
475 SolidBrush bgBrush(bgColor);
476
477 // +10% larger overall
478 int iconR = (int)(r * 1.1f); // 10% larger radius
479
480 // Crosshair lines (with gap for center circle)
481 int gap = (int)(9 * g_currentScale);
482 graphics.DrawLine(&pen, cx - iconR, cy, cx - gap, cy);
483 graphics.DrawLine(&pen, cx + gap, cy, cx + iconR, cy);
484 graphics.DrawLine(&pen, cx, cy - iconR, cx, cy - gap);
485 graphics.DrawLine(&pen, cx, cy + gap, cx, cy + iconR);
486
487 // Center circle with fill (AE anchor style)
488 int circleR = (int)(9 * g_currentScale);
489 graphics.FillEllipse(&bgBrush, cx - circleR, cy - circleR, circleR * 2,
490 circleR * 2);
491 graphics.DrawEllipse(&pen, cx - circleR, cy - circleR, circleR * 2,
492 circleR * 2);
493
494 // Draw preset number inside circle - scaled font
495 FontFamily fontFamily(L"Segoe UI");
496 Font font(&fontFamily, (int)(12 * g_currentScale), FontStyleBold,
497 UnitPixel);
498 SolidBrush textBrush(color);
499 StringFormat format;
500 format.SetAlignment(StringAlignmentCenter);
501 format.SetLineAlignment(StringAlignmentCenter);
502 wchar_t num[2] = {
503 static_cast<wchar_t>(L'1' + (type - NativeUI::OPT_CUSTOM_1)), 0};
504 RectF textRect((REAL)(cx - circleR), (REAL)(cy - circleR),
505 (REAL)(circleR * 2), (REAL)(circleR * 2));
506 graphics.DrawString(num, 1, &font, textRect, &format, &textBrush);
507 break;
508 }
509
510 case NativeUI::OPT_COMP_MODE: {
511 bool isCompMode = g_settings.useCompMode;
512 COLORREF colorRef =
513 hover ? COLOR_ICON_HOVER : (isCompMode ? COLOR_ORANGE : COLOR_BLUE);
514 Color color = toColor(colorRef);
515 Pen pen(color, 2.0f);
516 SolidBrush brush(color);
517
518 if (isCompMode) {
519 // Composition: wide rectangle with + in center
520 int w = r + 2;
521 int h = r - 2;
522 graphics.DrawRectangle(&pen, cx - w, cy - h, w * 2, h * 2);
523 // Plus sign
524 int ps = 3;
525 graphics.DrawLine(&pen, cx - ps, cy, cx + ps, cy);
526 graphics.DrawLine(&pen, cx, cy - ps, cx, cy + ps);
527 } else {
528 // Selection: square with corner squares (centered on corners)
529 graphics.DrawRectangle(&pen, cx - r, cy - r, r * 2, r * 2);
530 // Corner squares - centered on each corner
531 int cs = s + 1; // corner square size
532 int half = cs / 2;
533 graphics.FillRectangle(&brush, cx - r - half, cy - r - half, cs,
534 cs); // top-left
535 graphics.FillRectangle(&brush, cx + r - half, cy - r - half, cs,
536 cs); // top-right
537 graphics.FillRectangle(&brush, cx - r - half, cy + r - half, cs,
538 cs); // bottom-left
539 graphics.FillRectangle(&brush, cx + r - half, cy + r - half, cs,
540 cs); // bottom-right
541 }
542 break;
543 }
544
545 case NativeUI::OPT_MASK_MODE: {
546 bool maskOn = g_settings.useMaskRecognition;
547 COLORREF colorRef =
548 hover ? COLOR_ICON_HOVER : (maskOn ? COLOR_BLUE : COLOR_DARK_GRAY);
549 Color color = toColor(colorRef);
550 SolidBrush brush(color);
551 // Use hover background color when hovering for circle cutout
552 COLORREF bgColorRef = hover ? RGB(50, 60, 70) : COLOR_BG;
553 Color bgColor = toColor(bgColorRef);
554 SolidBrush bgBrush(bgColor);
555
556 // Horizontal rectangle (20% taller - 10% more than before)
557 int w = r + 3;
558 int h = (int)((r - 3) * 1.2f); // 20% taller total
559 graphics.FillRectangle(&brush, cx - w, cy - h, w * 2, h * 2);
560 // Circle cutout (30% larger than original)
561 int circleR = (int)(h / 2 * 1.3f); // 30% larger
562 graphics.FillEllipse(&bgBrush, cx - circleR, cy - circleR, circleR * 2,
563 circleR * 2);
564 break;
565 }
566
567 case NativeUI::OPT_SETTINGS: {
568 // Default: dark gray, hover: blue, active: blue
569 COLORREF colorRef =
570 hover ? COLOR_ICON_HOVER : (active ? COLOR_BLUE : COLOR_DARK_GRAY);
571 Color color = toColor(colorRef);
572 SolidBrush brush(color);
573
574 // User-style gear: 6 wide teeth, large center hole - 40% larger
575 int baseR = (int)(r * 1.4f); // 40% larger
576 int outerR = baseR; // Outer radius (teeth tips)
577 int midR = (int)(baseR * 0.7f); // Mid radius (teeth base) - 70%
578 int centerR = (int)(baseR * 0.35f); // Center hole - 35%
579
580 // Draw gear using polygon with 24 points (6 teeth, 4 points each)
581 Point gearPoints[24];
582 for (int i = 0; i < 6; i++) {
583 double baseAngle = i * 3.14159 * 2. / 6. - 3.14159 / 2.; // Start from top
584 double toothWidth = 3.14159 / 7.5; // Wider teeth
585
586 // Tooth left edge (inner to outer)
587 gearPoints[i * 4 + 0].X = cx + (int)(midR * cos(baseAngle - toothWidth));
588 gearPoints[i * 4 + 0].Y = cy + (int)(midR * sin(baseAngle - toothWidth));
589 // Tooth left tip
590 gearPoints[i * 4 + 1].X =
591 cx + (int)(outerR * cos(baseAngle - toothWidth / 2));
592 gearPoints[i * 4 + 1].Y =
593 cy + (int)(outerR * sin(baseAngle - toothWidth / 2));
594 // Tooth right tip
595 gearPoints[i * 4 + 2].X =
596 cx + (int)(outerR * cos(baseAngle + toothWidth / 2));
597 gearPoints[i * 4 + 2].Y =
598 cy + (int)(outerR * sin(baseAngle + toothWidth / 2));
599 // Tooth right edge (outer to inner)
600 gearPoints[i * 4 + 3].X = cx + (int)(midR * cos(baseAngle + toothWidth));
601 gearPoints[i * 4 + 3].Y = cy + (int)(midR * sin(baseAngle + toothWidth));
602 }
603 graphics.FillPolygon(&brush, gearPoints, 24);
604

606 graphics.FillEllipse(&brush, cx - midR, cy - midR, midR * 2, midR * 2);

610 Color bgColor = toColor(bgColorRef);
611 SolidBrush bgBrush(bgColor);
612 graphics.FillEllipse(&bgBrush, cx - centerR, cy - centerR, centerR * 2,
613 centerR * 2);
614 break;
615 }
616
617 case NativeUI::OPT_COPY_ANCHOR: {
618 // Orange if has clipboard data, blue if hover, else dark gray
619 COLORREF colorRef =
620 hover ? COLOR_ICON_HOVER
621 : (g_hasClipboardAnchor ? COLOR_ORANGE : COLOR_DARK_GRAY);
622 Color color = toColor(colorRef);
623 Pen pen(color, 2.0f);
624 SolidBrush brush(color);
625
626 // Two overlapping rectangles - 25% larger
627 int size = (int)(r * 1.25f) - 2;
628 int offset = 4;
629 // Back rectangle (outline only)
630 graphics.DrawRectangle(&pen, cx - size + offset, cy - size + offset,
631 size * 2 - offset * 2, size * 2 - offset * 2);
632 // Front rectangle (filled)
633 graphics.FillRectangle(&brush, cx - size, cy - size, size * 2 - offset * 2,
634 size * 2 - offset * 2);
635 break;
636 }

638 case NativeUI::OPT_PASTE_ANCHOR: {
639 // Dim if no clipboard data
640 COLORREF colorRef =
641 hover ? COLOR_ICON_HOVER
642 : (g_hasClipboardAnchor ? COLOR_BLUE : COLOR_DARK_GRAY);
643 Color color = toColor(colorRef);
644 Pen pen(color, 2.0f);
645 SolidBrush brush(color);

647 // Clipboard icon - smaller (was 1.2, now ~1.0)
648 int w = (int)(r * 1.0f) - 2;
649 int h = (int)(r * 1.0f) + 2;
650 // Main board
651 graphics.DrawRectangle(&pen, cx - w, cy - h + 4, w * 2, h * 2 - 4);
652 // Clip at top

713// Draw the grid with marks and glow using GDI+
714static void DrawGrid(HDC hdc) {
715 using namespace Gdiplus;
716
717 Graphics graphics(hdc);
718 graphics.SetSmoothingMode(SmoothingModeNone);
719 graphics.SetPixelOffsetMode(PixelOffsetModeHalf);
720
721 // Use cellSize directly (no spacing - grid lines will separate)
722 int cellTotal = g_config.cellSize; // No spacing now
723 // Apply grid offset for centered rectangular grids (with vertical padding)
724 int gridStartX = g_sidePanelWidth + g_scaledMargin + g_gridOffsetX;
725 int gridStartY = g_gridVerticalPadding + g_scaledMargin + g_gridOffsetY;
726 int radius = g_config.cellSize / 10; // Slightly larger dots
727 int hoverRadius = g_config.cellSize / 7; // Enhanced hover glow (was /8)
728 int len = (int)(cellTotal * 0.3); // Shorter marks (was 0.4)
729
730 bool compMode = g_settings.useCompMode;
731
732 // Helper to convert COLORREF to GDI+ Color
733 auto toColor = [](COLORREF c) {
734 return Color(255, GetRValue(c), GetGValue(c), GetBValue(c));
735 };
736
737 Color cellBgColor = toColor(COLOR_CELL_BG);
738
739 // Apply gridOpacity (mark opacity) to line/mark colors
740 BYTE markAlpha = (BYTE)(g_settings.gridOpacity * 255 / 100);
741 COLORREF lineColorRef = compMode ? COLOR_GRID_LINE_COMP : COLOR_GRID_LINE;
742 Color lineColor(markAlpha, GetRValue(lineColorRef), GetGValue(lineColorRef),
743 GetBValue(lineColorRef));
744
745 COLORREF glowInnerRef = compMode ? COLOR_GLOW_INNER_COMP : COLOR_GLOW_INNER;
746 Color glowInnerColor(255, GetRValue(glowInnerRef), GetGValue(glowInnerRef),
747 GetBValue(glowInnerRef)); // Glow stays full
748 COLORREF glowMidRef = compMode ? COLOR_GLOW_MID_COMP : COLOR_GLOW_MID;
749 Color glowMidColor(255, GetRValue(glowMidRef), GetGValue(glowMidRef),
750 GetBValue(glowMidRef));
751 COLORREF glowOuterRef = compMode ? COLOR_GLOW_OUTER_COMP : COLOR_GLOW_OUTER;
752 Color glowOuterColor(255, GetRValue(glowOuterRef), GetGValue(glowOuterRef),
753 GetBValue(glowOuterRef));
754
755 // Draw cell backgrounds with opacity
756 // cellOpacity: 0 = fully transparent (skip drawing), 100 = fully opaque
757 int gridWidth = g_config.gridWidth * cellTotal;
758 int gridHeight = g_config.gridHeight * cellTotal;
759
760 if (g_settings.cellOpacity > 0) {
761 // Linear alpha: 0->0, 100->255
762 BYTE cellAlpha = (BYTE)(g_settings.cellOpacity * 255 / 100);
763 Color cellBgWithAlpha(cellAlpha, GetRValue(COLOR_CELL_BG),
764 GetGValue(COLOR_CELL_BG), GetBValue(COLOR_CELL_BG));
765 SolidBrush cellBrush(cellBgWithAlpha);
766
767 // Fill entire grid area
768 graphics.FillRectangle(&cellBrush, gridStartX, gridStartY, gridWidth,
769 gridHeight);
770 }
771 // When cellOpacity is 0, skip drawing background entirely (transparent)
772
773 // Draw gray grid lines (separators between cells) - Enhanced visibility
774 Color gridLineColor(200, 100, 100, 100); // Brighter semi-transparent gray
775 Pen gridPen(gridLineColor, 1.0f);
776
777 // Vertical lines
778 for (int x = 1; x < g_config.gridWidth; x++) {
779 int lineX = gridStartX + x * cellTotal;
780 graphics.DrawLine(&gridPen, lineX, gridStartY, lineX,
781 gridStartY + gridHeight);
782 }
783 // Horizontal lines
784 for (int y = 1; y < g_config.gridHeight; y++) {
785 int lineY = gridStartY + y * cellTotal;
786 graphics.DrawLine(&gridPen, gridStartX, lineY, gridStartX + gridWidth,
787 lineY);
788 }
789
790 // Draw marks and dots using GDI+
791 Pen markPen(lineColor, 2.0f);
792 SolidBrush dotBrush(lineColor);
793
794 for (int y = 0; y < g_config.gridHeight; y++) {
795 for (int x = 0; x < g_config.gridWidth; x++) {
796 int cx = gridStartX + x * cellTotal + cellTotal / 2;
797 int cy = gridStartY + y * cellTotal + cellTotal / 2;
798
799 bool isHover = (x == g_hoverCellX && y == g_hoverCellY);
800
801 bool isLeft = (x == 0);
802 bool isRight = (x == g_config.gridWidth - 1);
803 bool isTop = (y == 0);
804 bool isBottom = (y == g_config.gridHeight - 1);
805 bool isCorner = (isLeft || isRight) && (isTop || isBottom);
806 bool isEdge = (isLeft || isRight || isTop || isBottom) && !isCorner;
807
808 // Reduced offset to move marks closer to center (was cellTotal/4)
809 int edgeOffset = cellTotal / 6;
810 int markX = cx, markY = cy;
811 if (isCorner || isEdge) {
812 if (isLeft)
813 markX -= edgeOffset;
814 if (isRight)
815 markX += edgeOffset;
816 if (isTop)
817 markY -= edgeOffset;
818 if (isBottom)
819 markY += edgeOffset;
820 }
821
822 // Different lengths for corner, edge, and center marks
823 int cornerLen = (int)(cellTotal * 0.35); // Longer for corners
824 int edgeLen = (int)(cellTotal * 0.25); // Shorter for edges
825 int centerLen = (int)(cellTotal * 0.25); // Same as edges for center
826
827 Color markColor = isHover ? glowInnerColor : lineColor;
828 Pen linePen(markColor, 2.0f);
829
830 if (isCorner) {
831 int L = cornerLen;
832 if (isTop && isLeft) {
833 graphics.DrawLine(&linePen, markX, markY + L, markX, markY);
834 graphics.DrawLine(&linePen, markX, markY, markX + L, markY);
835 } else if (isTop && isRight) {
836 graphics.DrawLine(&linePen, markX - L, markY, markX, markY);
837 graphics.DrawLine(&linePen, markX, markY, markX, markY + L);
838 } else if (isBottom && isLeft) {
839 graphics.DrawLine(&linePen, markX, markY - L, markX, markY);
840 graphics.DrawLine(&linePen, markX, markY, markX + L, markY);
841 } else {
842 graphics.DrawLine(&linePen, markX - L, markY, markX, markY);
843 graphics.DrawLine(&linePen, markX, markY, markX, markY - L);
844 }
845 } else if (isEdge) {
846 int L = edgeLen;
847 if (isTop) {
848 graphics.DrawLine(&linePen, markX - L, markY, markX + L, markY);
849 graphics.DrawLine(&linePen, markX, markY, markX, markY + L);
850 } else if (isBottom) {
851 graphics.DrawLine(&linePen, markX - L, markY, markX + L, markY);
852 graphics.DrawLine(&linePen, markX, markY - L, markX, markY);
853 } else if (isLeft) {
854 graphics.DrawLine(&linePen, markX, markY - L, markX, markY + L);
855 graphics.DrawLine(&linePen, markX, markY, markX + L, markY);
856 } else {
857 graphics.DrawLine(&linePen, markX, markY - L, markX, markY + L);
858 graphics.DrawLine(&linePen, markX - L, markY, markX, markY);
859 }
860 } else {
861 int L = centerLen;
862 graphics.DrawLine(&linePen, cx - L, cy, cx + L, cy);
863 graphics.DrawLine(&linePen, cx, cy - L, cx, cy + L);
864 }
865
866 int anchorX = (isCorner || isEdge) ? markX : cx;
867 int anchorY = (isCorner || isEdge) ? markY : cy;
868
869 // Draw center dot
870 SolidBrush dotBrush(lineColor);
871 graphics.FillEllipse(&dotBrush, anchorX - radius, anchorY - radius,
872 radius * 2, radius * 2);
873
874 // Draw hover glow
875 if (isHover) {
876 SolidBrush glowBrush3(glowOuterColor);
877 graphics.FillEllipse(&glowBrush3, anchorX - hoverRadius * 2,
878 anchorY - hoverRadius * 2, hoverRadius * 4,
879 hoverRadius * 4);
880
881 SolidBrush glowBrush2(glowMidColor);
882 graphics.FillEllipse(&glowBrush2, anchorX - hoverRadius - 3,
883 anchorY - hoverRadius - 3, (hoverRadius + 3) * 2,
884 (hoverRadius + 3) * 2);
885
886 SolidBrush glowBrush1(glowInnerColor);
887 graphics.FillEllipse(&glowBrush1, anchorX - hoverRadius,
888 anchorY - hoverRadius, hoverRadius * 2,
889 hoverRadius * 2);
890 }
891 }
892 }

897 LPARAM lParam) {

899 case WM_PAINT: {

911 FillRect(memDC, &rect, bgBrush);
912 DeleteObject(bgBrush);
913
914 DrawGrid(memDC);
915 DrawSidePanels(memDC);
916
917 BitBlt(hdc, 0, 0, rect.right, rect.bottom, memDC, 0, 0, SRCCOPY);
918
919 DeleteObject(memBitmap);
920 DeleteDC(memDC);
921
922 EndPaint(hwnd, &ps);
923 return 0;
924 }

927 POINT pt;
928 GetCursorPos(&pt);
929 NativeUI::UpdateHover(pt.x, pt.y);
930 return 0;
931 }
932
933 case WM_DESTROY:
934 g_gridWnd = NULL;
935 return 0;
936 }
937
938 return DefWindowProc(hwnd, msg, wParam, lParam);
939}
